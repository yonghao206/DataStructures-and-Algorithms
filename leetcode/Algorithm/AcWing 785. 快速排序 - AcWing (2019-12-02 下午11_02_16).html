AcWing 785. 快速排序

作者：   ddyyhh04 ,  2019-08-05 02:53:26 ,  阅读 22

   

0 


def quick_sort(nums, l, r):
    # 退出条件， l==r
    if l >= r:
        return
    x = nums[l]
    # 左右两个指针从两边切入
    i, j = l - 1, r + 1
    # 对于当前的list，只要i<j,就一直交换下去
    while i < j:

        while True:
            i += 1
            # 当左边的值大于等于x时，break
            if nums[i] >= x:
                break

        while True:
            j -= 1
            # 当右边的值小于等于x时，break
            if nums[j] <= x:
                break
        # 仍然i<j时，则交换二者
        if(i < j):
            nums[i], nums[j] = nums[j], nums[i]
    # 递归的去排左右两边
    quick_sort(nums, l, j)
    quick_sort(nums, j + 1, r)

if __name__ == "__main__":
    n = input()
    list_str = input()
    nums = list(map(int, list_str.split(' ')))
    quick_sort(nums, 0, len(nums) - 1)
    print(" ".join(map(str, nums)))
